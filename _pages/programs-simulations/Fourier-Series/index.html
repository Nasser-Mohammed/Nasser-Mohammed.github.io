\[ \begin{aligned} &f(t) = a_0 + \sum_{n=1}^{\infty}\left[ a_n \cos(\omega nt) + b_n \sin(\omega n t)\right]\\ \\ &a_0 = \frac{1}{T} \int_{-T/2}^{T/2} f(t)dt\\ \\ &a_n = \frac{2}{T}\int_{-\infty}^{\infty}f(t)\cos(2 \pi n t)\\ \\ &b_n = \frac{2}{T}\int_{-\infty}^{\infty}f(t)\sin(2 \pi n t)\\ \\ &\omega = \frac{2\pi}{T}, \text{ called the fundamental frequency}\\ \\ & \text{Where } T \text{ is the period of our function. The value such that } f(t) = f(t + T) \ \forall t \\ \end{aligned} \]
